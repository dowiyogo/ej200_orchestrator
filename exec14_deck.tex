\section{Resumen}

\begin{frame}{Resumen y puntos abiertos}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \textbf{Resultados principales:}
      \begin{enumerate}
        \item \textcolor{ej204col}{\textbf{EJ-204}} $\approx$
              \textcolor{ej230col}{\textbf{EJ-230}} en el centro
              ($141$ vs $139$\,ps);
              EJ-204 \textbf{mejor en bordes} ($365$ vs $419$\,ps a $x=690$)

        \item Componente F1 a $\sim4.7$\,ns en $x=690$ =
              \textcolor{okgreen}{\textbf{ToF geométrico}} del extremo lejano
              (QA-1c, $\delta=-5\%$). No es modo lento de centelleo.

        \item Top SiPM: señal robusta ($>20$\,PE), sensores del gap central
              altamente correlacionados ($r\approx0.88$)

        \item EJ-230 tiene $\tau_f$ más corto ($1.0$ vs $1.7$\,ns) pero mayor
              degradación posicional y coda residual en $\chi^2$
      \end{enumerate}
    \end{column}
    \begin{column}{0.44\textwidth}
      \textbf{\color{pendgold} Pendientes — confirmación Gerardo:}
      \begin{itemize}
        \item[$\square$] Topología SUM4:
              winner-takes-first vs suma de clusters
        \item[$\square$] Definición T4/T20:
              corte en $N_\text{pe}$ o eficiencia de trigger
        \item[$\square$] Jitter 20\,ps: conteo simple o doble (L+R)
        \item[$\square$] F3: impactos SiPM vs trayectorias en volumen
        \item[$\square$] Anticorrelación Par A ($r=-0.37$):
              geométrica o reflexiones
      \end{itemize}
      \vspace{0.4em}
      \textbf{Próximos pasos:}
      \begin{itemize}
        \item Medición $v_\text{eff}$ desde onset ToF ($\sim292$\,mm/ns, axial)
        \item Dataset EJ-230 EndTop (respuesta Top)
        \item Etapa 2: time-walk, readout jitter, electrónica real
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
